\begin{frame}
   {Raspberry-pi bonnet}
   \begin{itemize}
      \item Expansion boards for the raspberry-pi are called \textbf{hats}
      \item Since the rpi-zero is smaller, expansion boards for the zero
	      are called \textbf{bonnets}
   \end{itemize}
\end{frame}

\cprotect\note{

   There are many raspberry-pi zero bonnets available from such sources as Adafruit.

   \url{https://www.adafruit.com/category/929}
}
